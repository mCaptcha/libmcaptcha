23//!        .add_level(

26//!                .difficulty_factor(50)

28//!                .build()

30//!        )
31//!        .unwrap()

33//!            LevelBuilder::default()

35//!                .difficulty_factor(500)
36//!                .unwrap()
37//!                .build()
38//!                .unwrap(),
39//!        )
40//!        .unwrap()
41//!        .build()

45use crate::errors::*;
46use serde::{Deserialize, Serialize};

48/// Level struct that describes threshold-difficulty factor mapping
49#[derive(Debug, Deserialize, Serialize, Copy, Clone, PartialEq)]
50pub struct Level {
51    pub visitor_threshold: u32,
52    pub difficulty_factor: u32,
53}
54
55/// Bulder struct for [Level] to describe threshold-difficulty factor mapping
56#[derive(Debug, Copy, Clone, PartialEq)]
57pub struct LevelBuilder {
58    visitor_threshold: Option<u32>,
59    difficulty_factor: Option<u32>,
60}
61
62impl Default for LevelBuilder {
63    fn default() -> Self {
64        LevelBuilder {
65            visitor_threshold: None,
66            difficulty_factor: None,
67        }
68    }
69}
70
71impl LevelBuilder {
72    /// set visitor count for level
73    pub fn visitor_threshold(&mut self, visitor_threshold: u32) -> &mut Self {
74        self.visitor_threshold = Some(visitor_threshold);
75        self
76    }
77
78    /// set difficulty factor for level. difficulty_factor can't be zero because
79    /// Difficulty is calculated as:
80    /// ```no_run
81    /// let difficulty_factor = 500;
82    /// let difficulty = u128::max_value() - u128::max_value() / difficulty_factor;
83    /// ```
84    /// the higher the `difficulty_factor`, the higher the difficulty.
85    pub fn difficulty_factor(&mut self, difficulty_factor: u32) -> CaptchaResult<&mut Self> {
86        if difficulty_factor > 0 {
87            self.difficulty_factor = Some(difficulty_factor);
88            Ok(self)
89        } else {
90            Err(CaptchaError::DifficultyFactorZero)
91        }
92    }
93
94    /// build Level struct
95    pub fn build(&mut self) -> CaptchaResult<Level> {
96        if self.visitor_threshold.is_none() {
97            Err(CaptchaError::SetVisitorThreshold)
98        } else if self.difficulty_factor.is_none() {
99            Err(CaptchaError::SetDifficultyFactor)
100        } else {
101            Ok(Level {
102                difficulty_factor: self.difficulty_factor.unwrap(),
103                visitor_threshold: self.visitor_threshold.unwrap(),
104            })
105        }
106    }
107}

109/// Builder struct for [Defense]
110#[derive(Debug, Clone, PartialEq)]
111pub struct DefenseBuilder {
112    levels: Vec<Level>,
113}
114
115impl Default for DefenseBuilder {
116    fn default() -> Self {
117        DefenseBuilder { levels: vec![] }
118    }
119}
120
121impl DefenseBuilder {
122    /// add a level to [Defense]
123    pub fn add_level(&mut self, level: Level) -> CaptchaResult<&mut Self> {
124        for i in self.levels.iter() {
125            if i.visitor_threshold == level.visitor_threshold {
126                return Err(CaptchaError::DuplicateVisitorCount);
127            }
128        }
129        self.levels.push(level);
130        Ok(self)
131    }
132
133    /// Build [Defense]
134    pub fn build(&mut self) -> CaptchaResult<Defense> {
135        if !self.levels.is_empty() {
136            // sort levels to arrange in ascending order
137            self.levels.sort_by_key(|a| a.visitor_threshold);
138
139            for level in self.levels.iter() {
140                if level.difficulty_factor == 0 {
141                    return Err(CaptchaError::DifficultyFactorZero);
142                }
143            }
144
145            // as visitor count increases, difficulty_factor too should increse
146            // if it decreses, an error must be thrown
147            for i in 0..self.levels.len() - 1 {
148                if self.levels[i].difficulty_factor > self.levels[i + 1].difficulty_factor {
149                    return Err(CaptchaError::DecreaseingDifficultyFactor);
150                }
151            }
152
153            Ok(Defense {
154                levels: self.levels.to_owned(),
155                current_visitor_threshold: 0,
156            })
157        } else {
158            Err(CaptchaError::LevelEmpty)
159        }
160    }
161}

163/// struct describes all the different [Level]s at which an mCaptcha system operates
164#[derive(Debug, Serialize, Deserialize, Clone, PartialEq)]
165pub struct Defense {
166    levels: Vec<Level>,
167    // index of current visitor threshold
168    current_visitor_threshold: usize,
169}
170
171impl From<Defense> for Vec<Level> {
172    fn from(d: Defense) -> Self {
173        d.levels
174    }
175}
176
177impl Defense {
178    ///! Difficulty is calculated as:
179    ///! ```rust
180    ///! let difficulty = u128::max_value() - u128::max_value() / difficulty_factor;
181    ///! ```
182    ///! The higher the `difficulty_factor`, the higher the difficulty.
183
184    /// Get difficulty factor of current level of defense
185    pub fn get_difficulty(&self) -> u32 {
186        self.levels[self.current_visitor_threshold].difficulty_factor
187    }
188
189    /// tighten up defense. Increases defense level by a factor of one.
190    /// When defense is at max level, calling this method will have no effect
191    pub fn tighten_up(&mut self) {
192        if self.current_visitor_threshold < self.levels.len() - 1 {
193            self.current_visitor_threshold += 1;
194        }
195    }
196    /// Loosen up defense. Decreases defense level by a factor of one.
197    /// When defense is at the lowest level, calling this method will have no effect.
198    pub fn loosen_up(&mut self) {
199        if self.current_visitor_threshold > 0 {
200            self.current_visitor_threshold -= 1;
201        }
202    }
203
204    /// Set defense to maximum level
205    pub fn max_defense(&mut self) {
206        self.current_visitor_threshold = self.levels.len() - 1;
207    }
208
209    /// Set defense to minimum level
210    pub fn min_defense(&mut self) {
211        self.current_visitor_threshold = 0;
212    }
213
214    /// Get current level's  visitor threshold
215    pub fn current_level(&self) -> &Level {
216        &self.levels[self.current_visitor_threshold]
217    }
218
219    /// Get current level's  visitor threshold
220    pub fn visitor_threshold(&self) -> u32 {
221        self.levels[self.current_visitor_threshold].difficulty_factor
222    }
223}

229    #[test]
230    fn level_builder_works() {
231        let level = LevelBuilder::default()
232            .difficulty_factor(1)
233            .unwrap()
234            .visitor_threshold(0)
235            .build()
236            .unwrap();
237
238        assert_eq!(level.visitor_threshold, 0);
239        assert_eq!(level.difficulty_factor, 1);
240
241        assert_eq!(
242            LevelBuilder::default().difficulty_factor(0),
243            Err(CaptchaError::DifficultyFactorZero)
244        );
245
246        assert_eq!(
247            LevelBuilder::default()
248                .difficulty_factor(1)
249                .unwrap()
250                .build(),
251            Err(CaptchaError::SetVisitorThreshold)
252        );
253        assert_eq!(
254            LevelBuilder::default().visitor_threshold(10).build(),
255            Err(CaptchaError::SetDifficultyFactor)
256        );
257    }

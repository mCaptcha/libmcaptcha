18//! Redis Client/Connection manager that can handle both single and clustered Redis Instances
19use std::cell::RefCell;
20use std::rc::Rc;
21
22use redis::cluster::ClusterClient;
23use redis::Client;
24use redis::FromRedisValue;
25use redis::{aio::ConnectionManager, cluster::ClusterConnection};
26
27pub mod mcaptcha_redis;
28use crate::errors::*;
29
30/// Client configuration
31#[derive(Clone)]
32pub enum RedisConfig {
33    /// Redis server URL
34    Single(String),
35    /// List of URL of Redis nodes in cluster mode
36    Cluster(Vec<String>),
37}
38
39impl RedisConfig {
40    /// Create Redis connection
41    pub async fn connect(&self) -> RedisClient {
42        match self {
43            Self::Single(url) => {
44                let client = ConnectionManager::new(Client::open(url.as_str()).unwrap())
45                    .await
46                    .unwrap();
47                RedisClient::Single(client)
48            }
49            Self::Cluster(nodes) => {
50                let cluster_client = ClusterClient::new(nodes.to_owned()).unwrap();
51                RedisClient::Cluster(cluster_client)
52            }
53        }
54    }
55}
56
57/// Redis connection - manages both single and clustered deployments
58#[derive(Clone)]
59pub enum RedisConnection {
60    Single(Rc<RefCell<ConnectionManager>>),
61    Cluster(Rc<RefCell<ClusterConnection>>),
62}
63
64impl RedisConnection {
65    #[inline]
66    /// Get client. Uses interior mutability, so lookout for panics
67    pub fn get_client(&self) -> Self {
68        match self {
69            Self::Single(con) => Self::Single(Rc::clone(con)),
70            Self::Cluster(con) => Self::Cluster(Rc::clone(con)),
71        }
72    }
73    #[inline]
74    /// execute a redis command against a [Self]
75    pub async fn exec<T: FromRedisValue>(&self, cmd: &mut redis::Cmd) -> redis::RedisResult<T> {
76        match self {
77            RedisConnection::Single(con) => cmd.query_async(&mut *con.borrow_mut()).await,
78            RedisConnection::Cluster(con) => cmd.query(&mut *con.borrow_mut()),
79        }
80    }
81
82    pub async fn ping(&self) -> bool {
83        if let Ok(redis::Value::SimpleString(v)) = self.exec(&mut redis::cmd("PING")).await {
84            v == "PONG"
85        } else {
86            false
87        }
88    }
89}

91#[derive(Clone)]
92/// Client Configuration that can be used to get new connection shuld [RedisConnection] fail
93pub enum RedisClient {
94    Single(ConnectionManager),
95    Cluster(ClusterClient),
96}
97
98/// A Redis Client Object that encapsulates [RedisClient] and [RedisConnection].
99/// Use this when you need a Redis Client
100#[derive(Clone)]
101pub struct Redis {
102    _client: RedisClient,
103    connection: RedisConnection,
104}
105
106impl Redis {
107    /// create new [Redis]. Will try to connect to Redis instance specified in [RedisConfig]
108    pub async fn new(redis: RedisConfig) -> CaptchaResult<Self> {
109        let (_client, connection) = Self::connect(redis).await?;
110        let master = Self {
111            _client,
112            connection,
113        };
114        Ok(master)
115    }
116
117    /// Get client to do interact with Redis server.
118    ///
119    /// Uses Interior mutability so look out for panics
120    pub fn get_client(&self) -> RedisConnection {
121        self.connection.get_client()
122    }
123
124    async fn connect(redis: RedisConfig) -> CaptchaResult<(RedisClient, RedisConnection)> {
125        let redis = redis.connect().await;
126        let client = match &redis {
127            RedisClient::Single(c) => RedisConnection::Single(Rc::new(RefCell::new(c.clone()))),
128            RedisClient::Cluster(c) => {
129                let con = c.get_connection()?;
130                RedisConnection::Cluster(Rc::new(RefCell::new(con)))
131            }
132        };
133        Ok((redis, client))
134    }
135}

142    fn redis_url() -> &'static str {
143        static REDIS_URL: OnceLock<String> = OnceLock::new();
144        REDIS_URL.get_or_init(|| {
145            env::var("LIBMCAPTCHA_TEST_REDIS_URL").unwrap_or_else(|_| "redis://127.0.0.1".into())

147    }
148
149    #[actix_rt::test]
150    async fn ping_works() {
151        let r = Redis::new(RedisConfig::Single(redis_url().into()))
152            .await
153            .unwrap();
154        assert!(r.get_client().ping().await);
155    }
